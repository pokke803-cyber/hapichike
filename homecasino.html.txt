    function normalizeTodayTasks(savedTodayTasks, tasks) {
      const taskIds = new Set(tasks.map(task => task.id));
      if (Array.isArray(savedTodayTasks)) {
        return savedTodayTasks
          .filter(item => item && taskIds.has(item.taskId))
          .map(item => ({
            taskId: item.taskId,
            status: normalizeTaskStatus(item.status),
            startedAt: item.startedAt || "",
            completedAt: item.completedAt || "",
            ticketReward: Math.min(5, Math.max(1, Number(item.ticketReward) || Number(tasks.find(task => task.id === item.taskId)?.ticketReward) || 1)),
            ticketGranted: Boolean(item.ticketGranted)
          }));
      }

      const migrated = tasks
        .filter(task => normalizeTaskStatus(task.status, task.done) !== "idle")
        .map(task => {
          const status = normalizeTaskStatus(task.status, task.done);
          return {
            taskId: task.id,
            status,
            startedAt: "",
            completedAt: status === "done" ? todayString() : "",
            ticketReward: Math.min(5, Math.max(1, Number(task.ticketReward) || 1)),
            ticketGranted: status === "done"
          };
        });
      if (migrated.length) needsStateSave = true;
      return migrated;
    }

    function normalizeCategory(category, title = "", icon = "") {
      if (category === "creature" || category === "cooking" || category === "laundry" || category === "cleanup" || category === "outside") return category;
      if (category === "living") {
        needsStateSave = true;
        return "creature";
      }
      if (category === "other" || category === "study") {
        needsStateSave = true;
        return inferTaskCategory(title, icon);
      }
      needsStateSave = true;
      return inferTaskCategory(title, icon);
    }

    function normalizeTaskStatus(status, doneFlag = false) {

    }

      if (/おもちゃ|片づけ|片付け|そうじ|掃除|そうじき|へや|部屋|🧸|🧹|🧽/.test(text)) return "cleanup";
      return "cleanup";

    function normalizeTaskHistory(savedHistory, savedTasks) {
      const map = new Map();
      const sources = [
        ...(Array.isArray(savedHistory) ? savedHistory : []),
        ...(Array.isArray(savedTasks) ? savedTasks : [])
      ];

      sources.forEach((item, index) => {
        const title = item.title || item.name;
        if (!title) return;
        const normalized = {
          id: item.historyId || item.id || `history-${index}-${title}`,
          sourceTaskId: item.sourceTaskId || item.id || "",
          title,
          icon: item.icon || "⭐",
          category: normalizeCategory(item.category, title, item.icon),
          difficulty: ["easy", "normal", "hard"].includes(item.difficulty) ? item.difficulty : "easy",
          ticketReward: Math.min(5, Math.max(1, Number(item.ticketReward) || 1)),
          lastUsedAt: item.lastUsedAt || item.date || "",
          completedCount: Math.max(0, Number(item.completedCount) || (item.status === "done" || item.done ? 1 : 0))
        };
        const key = getTaskHistoryKey(normalized);
        const current = map.get(key);
        if (!current || (normalized.lastUsedAt || "") > (current.lastUsedAt || "")) {
          map.set(key, current ? { ...normalized, completedCount: Math.max(current.completedCount, normalized.completedCount) } : normalized);
        }
      });

      if (!Array.isArray(savedHistory) || map.size !== (savedHistory || []).length) needsStateSave = true;
      return Array.from(map.values()).sort(sortTaskHistory);
    }

    function getTaskHistoryKey(task) {
      return `${String(task.title || "").trim()}::${task.category || ""}`;
    }

    function sortTaskHistory(a, b) {
      if ((b.lastUsedAt || "") !== (a.lastUsedAt || "")) return (b.lastUsedAt || "").localeCompare(a.lastUsedAt || "");
      return (Number(b.completedCount) || 0) - (Number(a.completedCount) || 0);
    }

    function ensureAudioElements() {
      if (!bgmAudio) {
        bgmAudio = new Audio(audioSources.bgm);

        audio.preload = "auto";
        audio.volume = state.audioSettings.sfxVolume;
        sfxAudio[key] = audio;
      });
    }

    function getAudioContext() {
      if (!audioContext) {
        const AudioContextClass = window.AudioContext || window.webkitAudioContext;
        if (!AudioContextClass) return null;
        audioContext = new AudioContextClass();
      }
      if (audioContext.state === "suspended") audioContext.resume().catch(() => {});
      return audioContext;
    }

    function playTone(frequency, duration = .12, type = "triangle", volume = .05, delay = 0) {
      const context = getAudioContext();
      if (!context) return;
      const start = context.currentTime + delay;
      const osc = context.createOscillator();
      const gain = context.createGain();
      osc.type = type;
      osc.frequency.setValueAtTime(frequency, start);
      gain.gain.setValueAtTime(0.0001, start);
      gain.gain.exponentialRampToValueAtTime(volume, start + .02);
      gain.gain.exponentialRampToValueAtTime(0.0001, start + duration);
      osc.connect(gain);
      gain.connect(context.destination);
      osc.start(start);
      osc.stop(start + duration + .02);
    }

    function playFallbackSfx(name) {
      if (!state.audioSettings.sfxEnabled || !audioUnlocked) return;
      const patterns = {
        complete: [[523, .1], [659, .11, .08], [784, .14, .17]],
        ticket: [[740, .08], [988, .12, .09]],
        rouletteStart: [[196, .08], [247, .08, .08], [294, .08, .16]],
        rouletteStop: [[784, .09], [523, .12, .1]],
        success: [[659, .09], [784, .1, .08], [1046, .15, .18]],
        card: [[880, .1], [1175, .12, .1]],
        click: [[520, .045]],
        error: [[180, .12, 0, "sawtooth", .035]]
      };
      (patterns[name] || patterns.click).forEach(([freq, duration, delay = 0, type = "triangle", volume = .045]) => {
        playTone(freq, duration, type, volume * state.audioSettings.sfxVolume, delay);
      });
    }

    function startFallbackBgm() {
      if (bgmFallbackTimer || !state.audioSettings.bgmEnabled || !audioUnlocked) return;
      const notes = [392, 494, 587, 494, 440, 523, 659, 523];
      bgmFallbackTimer = setInterval(() => {
        if (!state.audioSettings.bgmEnabled || !audioUnlocked) return;
        playTone(notes[bgmFallbackStep % notes.length], .18, "sine", .018 * state.audioSettings.bgmVolume);
        if (bgmFallbackStep % 2 === 0) playTone(196, .08, "triangle", .01 * state.audioSettings.bgmVolume, .02);
        bgmFallbackStep += 1;
      }, 720);
    }

    function stopFallbackBgm() {
      if (!bgmFallbackTimer) return;
      clearInterval(bgmFallbackTimer);
      bgmFallbackTimer = null;
    }

    function startBgm() {
      if (!state.audioSettings.bgmEnabled || !audioUnlocked) return;
      ensureAudioElements();
      bgmAudio.volume = state.audioSettings.bgmVolume;
      const playPromise = bgmAudio.play();
      if (playPromise) {
        playPromise
          .then(() => stopFallbackBgm())
          .catch(() => startFallbackBgm());
      }
    }

    function stopBgm() {
      if (bgmAudio) bgmAudio.pause();
      stopFallbackBgm();
    }

    function unlockAudio() {
      if (audioUnlocked) return;
      audioUnlocked = true;
      ensureAudioElements();
      getAudioContext();
      if (state.audioSettings.bgmEnabled) startBgm();
      renderAudioControls();
    }

    function playSfx(name) {
      if (!state.audioSettings.sfxEnabled || !audioUnlocked) return;
      ensureAudioElements();
      const source = sfxAudio[name] || sfxAudio.click;
      if (!source) {
        playFallbackSfx(name);
        return;
      }
      const audio = source.cloneNode();
      audio.volume = state.audioSettings.sfxVolume;
      const playPromise = audio.play();
      if (playPromise) playPromise.catch(() => playFallbackSfx(name));
    }

    function isValidEmail(value) {
      const email = sanitizeEmail(value);
      return Boolean(email) && email.includes("@") && email.includes(".") && /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
    }

    function maskEmail(email) {
      const value = sanitizeEmail(email);
      const [name, domain] = value.split("@");
      if (!name || !domain) return value;
      const visible = name.slice(0, Math.min(3, name.length));
      return `${visible}${name.length > 3 ? "***" : ""}@${domain}`;
    }

    function upsertTaskHistory(task, options = {}) {
      if (!task || !task.title) return;
      const category = normalizeCategory(task.category, task.title, task.icon);
      const snapshot = {
        id: task.historyId || task.id || `history-${Date.now()}`,
        sourceTaskId: task.id || "",
        title: task.title,
        icon: task.icon || "⭐",
        category,
        difficulty: ["easy", "normal", "hard"].includes(task.difficulty) ? task.difficulty : "easy",
        ticketReward: Math.min(5, Math.max(1, Number(task.ticketReward) || 1)),
        lastUsedAt: nowIsoString(),
        completedCount: 0
      };
      const key = getTaskHistoryKey(snapshot);
      const existingIndex = state.taskHistory.findIndex(item => getTaskHistoryKey(item) === key);
      if (existingIndex >= 0) {
        const existing = state.taskHistory[existingIndex];
        state.taskHistory[existingIndex] = {
          ...existing,
          ...snapshot,
          id: existing.id || snapshot.id,
          sourceTaskId: existing.sourceTaskId || snapshot.sourceTaskId,
          completedCount: (Number(existing.completedCount) || 0) + (options.completed ? 1 : 0)
        };
      } else {
        state.taskHistory.push({
          ...snapshot,
          completedCount: options.completed ? 1 : 0
        });
      }
      state.taskHistory = state.taskHistory.sort(sortTaskHistory).slice(0, 80);
    }

    function renderTaskHistorySearch() {
      const list = document.getElementById("taskHistoryResults");
      const query = document.getElementById("taskHistorySearchInput").value.trim().toLowerCase();
      const source = [...state.taskHistory].sort(sortTaskHistory);
      const results = source.filter(item => {
        if (!query) return true;
        const searchable = [
          item.title,
          item.icon,
          getTaskCategoryLabel(item.category),
          difficultyLabels[item.difficulty],
          `${item.ticketReward}枚`
        ].join(" ").toLowerCase();
        return searchable.includes(query);
      });

      if (!results.length) {
        list.innerHTML = `<div class="log-item">りれきに見つからなかったよ。あたらしく追加してみよう。</div>`;
        return;

              <div class="history-result-meta">
                <span class="task-badge task-category">${escapeHtml(getTaskCategoryLabel(item.category))}</span>
                <span class="task-badge ${item.difficulty}">${difficultyLabels[item.difficulty]}</span>
                <span class="status-badge idle">🎫 ${Number(item.ticketReward) || 1}枚</span>
                <span class="status-badge done">回数 ${Number(item.completedCount) || 0}</span>
              </div>
            </div>
            <button class="history-add-button" data-history-add="${escapeHtml(item.id)}" ${added ? "disabled" : ""}>${added ? "追加済み" : "今日に追加"}</button>
          </article>

        button.addEventListener("click", () => addTaskFromHistory(button.dataset.historyAdd));
      });
    }

    function isTaskAlreadyToday(historyItem) {
      const key = getTaskHistoryKey(historyItem);
      return state.todayTasks.some(today => {
        const task = getTaskById(today.taskId);
        return task && getTaskHistoryKey(task) === key;
      });
    }

    function addTaskFromHistory(historyId) {
      const item = state.taskHistory.find(historyTask => historyTask.id === historyId);
      if (!item || isTaskAlreadyToday(item)) {
        renderTaskHistorySearch();
        return;
      }

      const key = getTaskHistoryKey(item);
      let task = state.tasks.find(candidate => getTaskHistoryKey(candidate) === key);
      if (!task) {
        task = {
          id: `task-${Date.now()}`,
          icon: item.icon || "⭐",
          title: item.title,
          difficulty: item.difficulty,
          ticketReward: Math.min(5, Math.max(1, Number(item.ticketReward) || 1)),
          category: normalizeCategory(item.category, item.title, item.icon),
          status: "idle",
          isDefault: false
        };
        state.tasks.push(task);
      }
      addTaskToToday(task.id, { silent: true });
      upsertTaskHistory(task);
      addHistory(`りれきから「${task.title}」を今日に追加しました。`);
      saveState();
      activeTaskCategory = "all";
      activeStatusTab = "active";
      render();
      renderTaskHistorySearch();
      showToast(`「${task.title}」を今日に追加しました。`);
    }

    function addTaskToToday(taskId, options = {}) {
      const task = getTaskById(taskId);
      if (!task || isTaskInTodayList(taskId)) return;
      state.todayTasks.push({
        taskId,
        status: "idle",
        startedAt: "",
        completedAt: "",
        ticketReward: Math.min(5, Math.max(1, Number(task.ticketReward) || 1)),
        ticketGranted: false
      });
      upsertTaskHistory(task);
      if (!options.silent) {
        addHistory(`「${task.title}」をやることリストに追加しました。`);
        saveState();
        activeStatusTab = "active";
        render();
        showToast(`「${task.title}」をやることに追加しました。`);
      }
    }

    function addSelectedTasksToToday() {
      const selectedIds = Array.from(bulkSelectedTaskIds);
      let addedCount = 0;
      selectedIds.forEach(taskId => {
        if (isTaskInTodayList(taskId)) return;

        if (state.todayTasks.length > beforeCount) addedCount += 1;

        showToast("追加できるお手伝いがありません。");
        renderBulkTaskPicker();

      saveState();
      bulkSelectedTaskIds = new Set();
      activeStatusTab = "active";
      closeBulkTaskPicker();

      playSfx("success");
      showToast(`${addedCount}つのお手伝いを追加しました！`);
    }

    function saveParentRewards() {
      const entries = Object.entries(parentRewardDraft);
      if (!entries.length) {
        document.getElementById("parentRewardMessage").textContent = "変更するチケット枚数を選んでください";
        return;
      }

      let changed = 0;
      entries.forEach(([taskId, reward]) => {
        const task = state.tasks.find(item => item.id === taskId);
        const nextReward = Number(reward);
        if (!task || !Number.isInteger(nextReward) || nextReward < 1 || nextReward > 5) return;
        if (task.ticketReward === nextReward) return;
        task.ticketReward = nextReward;
        upsertTaskHistory(task);
        changed += 1;
      });

      if (!changed) {
        document.getElementById("parentRewardMessage").textContent = "変更はありません";
        return;
      }

      parentRewardDraft = {};
      saveState();
      render();
      renderParentRewardSettings();
      document.getElementById("parentRewardMessage").textContent = "チケット枚数を変更しました";
      showToast("チケット枚数を変更しました。");
    }

    function executePayTransfer() {
      const fromId = document.getElementById("transferFromPay").value;
      const toId = document.getElementById("transferToPay").value;
      const amountValue = document.getElementById("transferAmountInput").value;
      const message = document.getElementById("transferMessage");
      const fromPay = payTypes.find(pay => pay.id === fromId);
      const toPay = payTypes.find(pay => pay.id === toId);
      const amount = Number(amountValue);

      message.className = "edit-error";
      if (!fromPay || !toPay) {
        message.textContent = "PAYを選んでください";
        return;
      }
      if (fromPay.id === toPay.id) {
        message.textContent = "振り替え元と振り替え先は別のPAYを選んでください";
        return;
      }
      if (!Number.isInteger(amount) || amount < 1) {
        message.textContent = "1以上の整数を入れてください";
        return;
      }

      const fromPoints = Number(state.points[fromPay.category]) || 0;
      if (amount > fromPoints) {
        message.textContent = "振り替え元のポイントがたりないよ";
        return;
      }

      const ok = window.confirm(`${fromPay.label} から ${toPay.label} へ ${amount}P 振り替えます。よろしいですか？`);
      if (!ok) return;

      state.points[fromPay.category] = fromPoints - amount;
      state.points[toPay.category] = (Number(state.points[toPay.category]) || 0) + amount;
      state.transferHistory.unshift({
        id: `transfer-${Date.now()}`,
        date: todayString(),
        fromId: fromPay.id,
        fromLabel: fromPay.label,
        toId: toPay.id,
        toLabel: toPay.label,
        amount
      });
      state.transferHistory = state.transferHistory.slice(0, 50);
      addHistory(`${fromPay.label} から ${toPay.label} に ${amount}P 振り替えました。`);
      saveState();
      document.getElementById("transferAmountInput").value = "";
      message.textContent = `${fromPay.label} から ${toPay.label} に ${amount}P 振り替えました`;
      message.className = "edit-error pay-message success";
      render();
      renderPayTransferPanel();
      playSfx("success");
      showToast("PAYポイントを振り替えました。");
    }

    function saveParentTaskForm(event) {
      event.preventDefault();
      const title = document.getElementById("parentTaskTitleInput").value.trim();
      const icon = document.getElementById("parentTaskIconInput").value.trim() || "🎟️";
      const category = document.getElementById("parentTaskCategoryInput").value;
      const difficulty = document.getElementById("parentTaskDifficultyInput").value;
      const reward = Number(parentTaskRewardDraft);
      const error = document.getElementById("parentTaskFormError");

      if (!title) {
        error.textContent = "お手伝い名を入れてください";
        return;
      }
      if (!taskCategories.some(item => item.id === category && item.id !== "all")) {
        error.textContent = "カテゴリーを選んでください";
        return;
      }
      if (!["easy", "normal", "hard"].includes(difficulty)) {
        error.textContent = "難易度を選んでください";
        return;
      }
      if (!Number.isInteger(reward) || reward < 1 || reward > 5) {
        error.textContent = "TICKET枚数は1〜5枚で選んでください";
        return;
      }

      const existing = state.tasks.find(task => task.id === parentEditingTaskId);
      if (existing) {
        existing.title = title;
        existing.icon = icon;
        existing.category = category;
        existing.difficulty = difficulty;
        existing.ticketReward = reward;
        upsertTaskHistory(existing);
        showToast("お手伝いを保存しました。");
      } else {
        const task = {
          id: `task-${Date.now()}`,
          title,
          icon,
          category,
          difficulty,
          ticketReward: reward,
          status: "idle",
          isDefault: false
        };
        state.tasks.push(task);
        upsertTaskHistory(task);
        showToast("お手伝いを追加しました。");
      }

      saveState();
      render();
      openParentTaskEditPage();
    }

    function renderWheel() {
      const wheel = document.getElementById("wheel");
      wheel.querySelectorAll(".wheel-label").forEach(label => label.remove());
      const segments = getCurrentSegments();
      const totalWeight = segments.reduce((sum, segment) => sum + segment.weight, 0);
      let currentDeg = 0;
      const gradients = [];

      segments.forEach((segment, index) => {
        const size = segment.weight / totalWeight * 360;
        const endDeg = currentDeg + size;
        gradients.push(`${segment.color} ${currentDeg.toFixed(2)}deg ${endDeg.toFixed(2)}deg`);
        const label = document.createElement("div");
        label.className = "wheel-label";
        const labelDeg = currentDeg + size / 2 - 90;
        label.style.transform = `rotate(${labelDeg.toFixed(2)}deg) translate(-4px, -50%)`;
        label.innerHTML = `${segment.icon}<br>${escapeHtml(segment.label)}`;
        wheel.appendChild(label);
        currentDeg = endDeg;
      });
      wheel.style.background = `conic-gradient(${gradients.join(", ")})`;
    }

    function weightedPick(segments) {
      const totalWeight = segments.reduce((sum, segment) => sum + segment.weight, 0);
      let roll = Math.random() * totalWeight;
      let startWeight = 0;
      for (const segment of segments) {
        if (roll < segment.weight) {
          return { segment, startWeight, totalWeight };
        }
        roll -= segment.weight;
        startWeight += segment.weight;
      }
      return { segment: segments[segments.length - 1], startWeight: totalWeight - segments[segments.length - 1].weight, totalWeight };
    }

    function spinWheel() {
      if (spinning) return;
      if (state.coins < activeBet) {
        playSfx("error");
        showToast(`チケットがあと ${activeBet - state.coins} まい必要です。お手伝いで集めよう。`);
        return;
      }
      spinning = true;
      playSfx("rouletteStart");
      latestRouletteResult = null;
      const useHappyChanceBonus = happyChanceReserved && state.happyChanceCards > 0;
      if (useHappyChanceBonus) {
        state.happyChanceCards -= 1;
        happyChanceReserved = false;
      }
      state.coins -= activeBet;
      updateStats();
      const segments = getCurrentSegments();
      const picked = weightedPick(segments);
      const segmentSize = picked.segment.weight / picked.totalWeight * 360;
      const startDeg = picked.startWeight / picked.totalWeight * 360;
      const centerDeg = startDeg + segmentSize / 2;
      const target = 360 - centerDeg;
      const currentMod = ((rotation % 360) + 360) % 360;
      rotation += 1440 + ((target - currentMod + 360) % 360);
      document.getElementById("wheel").style.transform = `rotate(${rotation}deg)`;
      setTimeout(() => {
        const category = categories[activeCategory];
        const before = state.points[activeCategory];
        const basePoints = picked.segment.type === "happyChance" ? 0 : picked.segment.points;
        const bonusPoints = useHappyChanceBonus ? 1 : 0;
        const totalPoints = basePoints + bonusPoints;
        const after = before + totalPoints;
        const actualChange = after - before;
        state.points[activeCategory] = after;
        const ticketBonus = picked.segment.points === 0 && picked.segment.type !== "happyChance" ? 1 : 0;
        if (ticketBonus) state.coins += ticketBonus;
        if (picked.segment.type === "happyChance") state.happyChanceCards += 1;
        latestRouletteResult = {
          category: activeCategory,
          used: false,
          canUseCard: false
        };
        const resultBox = document.getElementById("luckyText");
        resultBox.className = useHappyChanceBonus ? "result-good" : getResultClass(picked.segment);
        const bonusMessage = useHappyChanceBonus
          ? `<div class="segment-chip chance"><span>ハピチャンカード</span><span>💖 +1P</span></div>`
          : "";
        resultBox.innerHTML = `
          <span>${category.emoji} ${getResultMessage(category, picked.segment, before, after, bonusPoints)}</span>
          ${createResultSegmentHtml(picked.segment)}
          ${bonusMessage}
        `;
        const chanceText = picked.segment.type === "happyChance" ? " / ハピチャンカード +1" : "";
        const bonusText = useHappyChanceBonus ? " / ハピチャンカード使用 +1P" : "";
        addHistory(`${category.label}ルーレット BET${activeBet}: ${picked.segment.label}（実際の変化 ${formatPoints(actualChange)}${ticketBonus ? " / チケット +1" : ""}${chanceText}${bonusText}）`);
        saveState();
        spinning = false;
        render();
        playSfx("rouletteStop");
        if (picked.segment.points > 0 || useHappyChanceBonus) burstConfetti();
        if (useHappyChanceBonus && basePoints === 0) {
          playSfx("card");
          showToast("0Pだったけど、カードで +1P！");
        } else if (useHappyChanceBonus) {
          playSfx("card");
          showToast("ハピチャンカードを使ったよ！+1P！");
        } else if (picked.segment.type === "happyChance") {
          burstConfetti();
          playSfx("card");
          showToast("ハピチャンカードを1枚ゲット！次回+1Pできるよ！");
        } else if (picked.segment.points > 0) {
          playSfx(picked.segment.points >= 10 ? "success" : "ticket");
          showToast(`${category.pointName} ${formatPoints(picked.segment.points)}！`);
        } else {
          playSfx("ticket");
          showToast("ポイントはそのまま！チケットが1まい戻ってきたよ。");
        }
      }, 3700);
    }

    function usePayPoints(payId) {
      const pay = payTypes.find(item => item.id === payId);
      if (!pay) return;
      const amount = Number(usePayAmount[payId]);
      const current = state.points[pay.category] || 0;

      if (!usePayAmount[payId]) {
        playSfx("error");
        setPayMessage(payId, "つかうポイントを入力してね", "error");
        return;
      }

      if (!Number.isInteger(amount) || amount <= 0) {
        playSfx("error");
        setPayMessage(payId, "1以上の数字を入れてね", "error");
        return;
      }

      if (amount > current) {
        playSfx("error");
        setPayMessage(payId, "ポイントがたりないよ", "error");
        return;
      }

      const nextPoints = current - amount;
      state.points[pay.category] = nextPoints;
      addPayLog(pay, amount, current, nextPoints);
      usePayAmount[payId] = "";
      addHistory(`${pay.label}を${amount}Pつかいました。`);
      saveState();
      render();
      setPayMessage(payId, `${pay.label}を${amount}Pつかったよ！`, "success");
      burstConfetti();
      playSfx("success");
      showToast(`${pay.label}を${amount}Pつかったよ！`);
    }
